data Graph (a : Type) where
    EmptyG  : Graph a
  ; Vertex  : a -> Graph a
  ; Overlay : (Graph a) -> (Graph a) -> Graph a
  ; Connect : (Graph a) -> (Graph a) -> Graph a

         . (List a) -> Graph a

edge : forall {a : Type} 
     . a -> a -> Graph a

     . ((a -> b) [0..∞]) -> (Graph a) -> Graph b
