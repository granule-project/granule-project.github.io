data Fin : Nat -> Type where
    FZ : forall {k : Nat} . Fin (k + 1);
    FS : forall {k : Nat} . Fin k -> Fin (k + 1)

upcast : forall {n : Nat, m : Nat} . {m .≤ n} =>
    Fin m -> Fin n

fromN : forall {n : Nat, m : Nat} . N n -> Fin (n + 1)

fromNInto : forall {n : Nat, m : Nat} . {m ≥ 1} =>
    N n -> Fin (n + m)
